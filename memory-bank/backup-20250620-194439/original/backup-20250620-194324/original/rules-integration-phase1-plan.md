# –®–∞–≥ 4: –ê–Ω–∞–ª–∏–∑ —Ä–µ–∑—É–ª—å—Ç–∞—Ç–æ–≤ —Ç–µ—Å—Ç–æ–≤
analyze_test_results() {
    local test_output="$1"
    local analysis_file="qa-analysis/test-results/analysis.log"

    echo "=== QA –ê–ù–ê–õ–ò–ó –†–ï–ó–£–õ–¨–¢–ê–¢–û–í ===" > "$analysis_file"
    echo "–î–∞—Ç–∞: $(date)" >> "$analysis_file"
    echo "" >> "$analysis_file"

    # –û–±—â–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞
    local total_tests=$(grep -E "(pass|fail)" "$test_output" | wc -l)
    local passed_tests=$(grep "(pass)" "$test_output" | wc -l)
    local failed_tests=$(grep "(fail)" "$test_output" | wc -l)
    local success_rate=$((passed_tests * 100 / total_tests))

    echo "–û–±—â–∞—è —Å—Ç–∞—Ç–∏—Å—Ç–∏–∫–∞:" >> "$analysis_file"
    echo "- –í—Å–µ–≥–æ —Ç–µ—Å—Ç–æ–≤: $total_tests" >> "$analysis_file"
    echo "- –ü—Ä–æ—à–ª–∏: $passed_tests" >> "$analysis_file"
    echo "- –£–ø–∞–ª–∏: $failed_tests" >> "$analysis_file"
    echo "- –£—Å–ø–µ—à–Ω–æ—Å—Ç—å: $success_rate%" >> "$analysis_file"
    echo "" >> "$analysis_file"

    # –û–ø—Ä–µ–¥–µ–ª–µ–Ω–∏–µ –∫—Ä–∏—Ç–∏—á–Ω–æ—Å—Ç–∏
    if [ "$success_rate" -lt 80 ]; then
        echo "üö® –ö–†–ò–¢–ò–ß–ï–°–ö–û–ï –°–û–°–¢–û–Ø–ù–ò–ï: –£—Å–ø–µ—à–Ω–æ—Å—Ç—å < 80%" >> "$analysis_file"
        echo "qa_status=CRITICAL" > qa-analysis/status.env
    elif [ "$success_rate" -lt 95 ]; then
        echo "‚ö†Ô∏è –¢–†–ï–ë–£–ï–¢ –í–ù–ò–ú–ê–ù–ò–Ø: –£—Å–ø–µ—à–Ω–æ—Å—Ç—å < 95%" >> "$analysis_file"
        echo "qa_status=WARNING" > qa-analysis/status.env
    else
        echo "‚úÖ –•–û–†–û–®–ï–ï –°–û–°–¢–û–Ø–ù–ò–ï: –£—Å–ø–µ—à–Ω–æ—Å—Ç—å >= 95%" >> "$analysis_file"
        echo "qa_status=GOOD" > qa-analysis/status.env
    fi
}
